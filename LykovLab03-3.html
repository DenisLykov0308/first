Програмування

В мене були спроби створити власну гру ще у 13 років, але доробити її до кінця не зміг. На 2-му курсі в мене була курсова робота, мета якої була розробка гри на мові C#, використовуючи будь-яке ядро для неї. Тому я б хотів розмістити частину коду для реалізації атаки гравця:

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PlayerAttack : MonoBehaviour
{
    // Аніматор для запуску анімацій
    public Animator anim;
    // Точка атаки, від якої буде перевірятися зона ураження
    public Transform attakPoint;
    // Радіус зони ураження
    public float attakRange = 0.5f;
    // Шар ворогів, які будуть уражені
    public LayerMask Enemy;
    // Шкода, яка буде завдана при атаці
    public int atkDamage = 1;

    void Update()
    {
        // Перевірка на натискання лівої кнопки миші для запуску атаки
        if (Input.GetKeyDown(KeyCode.Mouse0))
        {
            Attack();
        }
    }

    void Attack()
    {
        // Запускаємо анімацію атаки
        anim.SetTrigger("attak");
        // Знаходимо всі об'єкти у зоні атаки, які мають шар ворогів
        Collider2D[] hitenemies = Physics2D.OverlapCircleAll(attakPoint.position,
        attakRange, Enemy);
        // Проходимося по всіх уражених ворогах
        foreach(Collider2D Enemy in hitenemies)
        {
            // Завдаємо шкоди кожному ворогу, який має компонент Enemy
            Enemy.GetComponent().takedamage(atkDamage);
        }
    }
    
    // Відображення зони ураження в редакторі для зручності налаштування
    private void OnDrawGizmosSelected()
    {
        if (attakPoint == null)
        {
            return;
        }
        // Малюємо сферу в місці точки атаки для візуалізації зони ураження
        Gizmos.DrawWireSphere(attakPoint.position, attakRange);
    }
}
